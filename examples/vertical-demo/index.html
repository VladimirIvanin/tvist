Код для галереи

const thumbs = new Tvist('#thumbs', {
  direction: 'vertical',
  slideMinSize: 80, // Адаптивная высота миниатюр
  gap: 10,
  isNavigation: true
})

const main = new Tvist('#main', {
  direction: 'horizontal', // По умолчанию
  perPage: 1
})

main.sync(thumbs) // Синхронизация
